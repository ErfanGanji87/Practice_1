void math()
{
    float ax = 0;
    float bx = 0;
    float c = 0;

    cout << endl <<"Enter Parameter: ";
    cin >> ax >> bx >> c;

    float d = pow(bx, 2) - (4 * ax * c);

    float a1 = 0;
    float a2 = 0;

    float r = 0;
    float i = 0;


    if (d >= 0)
    {
        a1 = ((-bx) + sqrt(d)) / (2 * ax);
        a2 = ((-bx) - sqrt(d)) / (2 * ax);

        cout << "A1: " << a1 << endl;
        cout << "A2: " << a2 << endl;
    }
    else
    {
        r = (-bx) / (2 * ax);
        i = sqrt(-d) / (2 * ax);

        cout << "A: " << r << "+-" << i << endl;
    }
}
